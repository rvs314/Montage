598 template <typename T>
599 void
600 vector<T>::assign(size_type count, const_reference value)
601 {
602  pool_base pb = get_pool();
603 
604  transaction::run(pb, [&] {
605  if (count <= capacity()) {
606  /*
607  * Reallocation is not needed. First, replace old
608  * elements with new ones in range [0, size()).
609  * Depending on count, either call remaining old
610  * elements destructors, or append more new elements.
611  */
612  size_type size_old = _size;
613  snapshot_data(0, size_old);
614 
615  std::fill_n(
616  &_data[0],
617  (std::min)(count,
618  static_cast<size_type>(size_old)),
619  value);
620 
621  if (count > size_old) {
622 #if LIBPMEMOBJ_CPP_VG_PMEMCHECK_ENABLED
623  /*
624  * Range of memory:
625  * [&_data[size_old], &_data[count])
626  * is undefined, there is no need to snapshot
627  * and eventually rollback old data.
628  */
629  VALGRIND_PMC_ADD_TO_TX(
630  &_data[static_cast<difference_type>(
631  size_old)],
632  sizeof(T) * (count - size_old));
633 #endif
634 
635  construct(size_old, count - size_old, value);
636  /*
637  * XXX: explicit persist is required here
638  * because given range wasn't snapshotted and
639  * won't be persisted automatically on tx
640  * commit. This can be changed once we will have
641  * implemented "uninitialized" flag for
642  * pmemobj_tx_xadd in libpmemobj.
643  */
644  pb.persist(&_data[static_cast<difference_type>(
645  size_old)],
646  sizeof(T) * (count - size_old));
647  } else {
648  shrink(count);
649  }
650  } else {
651  dealloc();
652  alloc(count);
653  construct(0, count, value);
654  }
655  });
656 }

676 template <typename T>
677 template <typename InputIt,
678  typename std::enable_if<detail::is_input_iterator<InputIt>::value,
679  InputIt>::type *>
680 void
681 vector<T>::assign(InputIt first, InputIt last)
682 {
683  pool_base pb = get_pool();
684 
685  size_type size_new = static_cast<size_type>(std::distance(first, last));
686 
687  transaction::run(pb, [&] {
688  if (size_new <= capacity()) {
689  /*
690  * Reallocation is not needed. First, replace old
691  * elements with new ones in range [0, size()).
692  * Depending on size_new, either call remaining old
693  * elements destructors, or append more new elements.
694  */
695  size_type size_old = _size;
696  snapshot_data(0, size_old);
697 
698  InputIt mid = last;
699  bool growing = size_new > size_old;
700 
701  if (growing) {
702 #if LIBPMEMOBJ_CPP_VG_PMEMCHECK_ENABLED
703  /*
704  * Range of memory:
705  * [&_data[size_old], &_data[size_new])
706  * is undefined, there is no need to snapshot
707  * and eventually rollback old data.
708  */
709  VALGRIND_PMC_ADD_TO_TX(
710  &_data[static_cast<difference_type>(
711  size_old)],
712  sizeof(T) * (size_new - size_old));
713 #endif
714 
715  mid = first;
716  std::advance(mid, size_old);
717  }
718 
719  iterator shrink_to = std::copy(first, mid, &_data[0]);
720 
721  if (growing) {
722  construct_range_copy(size_old, mid, last);
723  /*
724  * XXX: explicit persist is required here
725  * because given range wasn't snapshotted and
726  * won't be persisted automatically on tx
727  * commit. This can be changed once we will have
728  * implemented "uninitialized" flag for
729  * pmemobj_tx_xadd in libpmemobj.
730  */
731  pb.persist(&_data[static_cast<difference_type>(
732  size_old)],
733  sizeof(T) * (size_new - size_old));
734  } else {
735  shrink(static_cast<size_type>(std::distance(
736  iterator(&_data[0]), shrink_to)));
737  }
738  } else {
739  dealloc();
740  alloc(size_new);
741  construct_range_copy(0, first, last);
742  }
743  });
744 }

2294 void
2295 vector<T>::insert_gap(size_type idx, size_type count)
2296 {
2297  assert(pmemobj_tx_stage() == TX_STAGE_WORK);
2298 
2299  if (_capacity >= _size + count) {
2300  pointer dest =
2301  &_data[static_cast<difference_type>(size() + count)];
2302  pointer begin = &_data[static_cast<difference_type>(idx)];
2303  pointer end = &_data[static_cast<difference_type>(size())];
2304 
2305  /*

2307  * we can optimize it by calling:
2308  * transaction::snapshot<T>(begin, size() - idx).

2310  * now pmemcheck will report an error (uninitialized part of

2312  *
2313  * XXX: future optimization: we don't have to snapshot data
2314  * which we will not overwrite

2317  VALGRIND_MAKE_MEM_DEFINED(end, sizeof(T) * count);
2318 #endif

2320 
2321  std::move_backward(begin, end, dest);
2322  } else {
2323  /*
2324  * XXX: future optimization: we don't have to snapshot data
2325  * which we will not overwrite
2326  */
2327  snapshot_data(0, _size);
2328 
2329  auto old_data = _data;
2330  auto old_size = _size;
2331  pointer old_begin = &_data[0];
2332  pointer old_mid = &_data[static_cast<difference_type>(idx)];

2341  construct_range(idx + count, old_mid, old_end);

2346  old_data[static_cast<difference_type>(i)]);
