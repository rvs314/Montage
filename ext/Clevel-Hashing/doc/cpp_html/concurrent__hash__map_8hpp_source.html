177 class atomic_backoff {
183  static const int32_t LOOPS_BEFORE_YIELD = 16;
184  int32_t count;
185 
186  static inline void
187  __pause(int32_t delay)
188  {
189  for (; delay > 0; --delay) {
190 #if _MSC_VER
191  YieldProcessor();
192 #elif __GNUC__ && (__i386__ || __x86_64__)
193  // Only i386 and x86-64 have pause instruction
194  __builtin_ia32_pause();
195 #endif
196  }
197  }
198 
199 public:
203  atomic_backoff(const atomic_backoff &) = delete;
207  atomic_backoff &operator=(const atomic_backoff &) = delete;
208 
210  /* In many cases, an object of this type is initialized eagerly on hot
211  * path, as in for(atomic_backoff b; ; b.pause()) {...} For this reason,
212  * the construction cost must be very small! */
213  atomic_backoff() : count(1)
214  {
215  }
216 
220  atomic_backoff(bool) : count(1)
221  {
222  pause();
223  }
224 
228  void
229  pause()
230  {
231  if (count <= LOOPS_BEFORE_YIELD) {
232  __pause(count);
233  /* Pause twice as long the next time. */
234  count *= 2;
235  } else {
236  /* Pause is so long that we might as well yield CPU to
237  * scheduler. */
238  std::this_thread::yield();
239  }
240  }
241 
245  bool
246  bounded_pause()
247  {
248  __pause(count);
249  if (count < LOOPS_BEFORE_YIELD) {
250  /* Pause twice as long the next time. */
251  count *= 2;
252  return true;
253  } else {
254  return false;
255  }
256  }
257 
258  void
259  reset()
260  {
261  count = 1;
262  }

484 template <typename Bucket>
485 class segment_traits {
486 public:
488  using segment_index_t = size_t;
489  using size_type = size_t;
490  using bucket_type = Bucket;
491 
492 protected:
494  constexpr static size_type max_allocation_size = PMEMOBJ_MAX_ALLOC_SIZE;
495 
497  constexpr static segment_index_t first_big_block = 27;
498  /* TODO: avoid hardcoded value; need constexpr similar to:
499  * Log2(max_allocation_size / sizeof(bucket_type)) */
500 
502  constexpr static size_type big_block_size = size_type(1)
503  << first_big_block;
504 
505  /* Block size in bytes cannot exceed max_allocation_size */
506  static_assert((big_block_size * sizeof(bucket_type)) <
507  max_allocation_size,
508  "Block size exceeds max_allocation_size");
509 
511  constexpr static segment_index_t
512  first_block_in_segment(segment_index_t seg)
513  {
514  return seg < first_big_block
515  ? seg
516  : (first_big_block +
517  (segment_index_t(1) << (seg - first_big_block)) - 1);
518  }
519 
521  constexpr static size_type
522  blocks_in_segment(segment_index_t seg)
523  {
524  return seg < first_big_block
525  ? segment_index_t(1)
526  : segment_index_t(1) << (seg - first_big_block);
527  }
528 
530  constexpr static size_type
531  block_size(segment_index_t b)
532  {
533  return b < first_big_block ? segment_size(b ? b : 1)
534  : big_block_size;
535  }
536 
537 public:
539  constexpr static segment_index_t embedded_segments = 1;
540 
542  constexpr static size_type embedded_buckets = 1 << embedded_segments;
543 
545  constexpr static segment_index_t number_of_segments = 32;
546 
548  static const size_type first_block = 8;
549 
551  constexpr static segment_index_t
552  number_of_blocks()
553  {
554  return first_block_in_segment(number_of_segments);
555  }
556 
558  static segment_index_t
559  segment_index_of(size_type index)
560  {
561  return segment_index_t(Log2(index | 1));
562  }
563 
565  constexpr static segment_index_t
566  segment_base(segment_index_t k)
567  {
568  return (segment_index_t(1) << k) & ~segment_index_t(1);
569  }
570 
572  constexpr static size_type
573  segment_size(segment_index_t k)
574  {
575  return size_type(1) << k; // fake value for k == 0
576  }
577  static_assert(
578  embedded_segments < first_big_block,
579  "Number of embedded segments cannot exceed max_allocation_size");
580 }; /* End of class segment_traits */

1994  !b_old.upgrade_to_writer()) {

1997  * concurrent erase */
